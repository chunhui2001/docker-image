
### 显示所有已安装的组件
$ npm list -g --depth=0

### 安装 truffle
$ npm install -g truffle

### 编译
$ truffle compile

### 部署
$ truffle migrate
OR
$ truffle migrate --reset
OR 
$ truffle migrate --network=development
OR 
$ truffle migrate --reset --network=development --config=MyERC20/truffle-config-docker.js
